  <input type="number" id="amount">

  <label>Годовая ставка (%)</label>
  <input type="number" id="rate">

  <label>Срок (в месяцах)</label>
  <input type="number" id="months">

  <button onclick="calculate()">Посчитать</button>

  <div id="output"></div>
</div>

<script>
function calculate() {
  let P = parseFloat(document.getElementById("amount").value);
  let r = parseFloat(document.getElementById("rate").value) / 100 / 12;
  let n = parseInt(document.getElementById("months").value);

  if (!P || !r || !n) {
    alert("Заполните все поля");
    return;
  }

  let payment = P * r / (1 - Math.pow(1 + r, -n));
  let total = payment * n;
  let overpay = total - P;

  document.getElementById("output").innerHTML = `
    <div class="result">
      <b>Ежемесячный платёж:</b> ${payment.toFixed(2)}<br>
      <b>Переплата:</b> ${overpay.toFixed(2)}
    </div>

    <div class="paywall">
      <b>Хотите подробный расчёт?</b><br>
      Таблица платежей, сравнение вариантов и PDF — после оплаты.
    </div>
  `;
}
</script>
